Fields

"Types" of turtles can remember things via fields.

class GiveUpTurtle extends Turtle
  constructor: ->
    super
    @timesMoved = 0 # Make a new field called "timesMoved"

  fd: ->
    @timesMoved += 1
    if @timesMoved < 3
      super

a = new GiveUpTurtle()
a.pen red
for [1..10]
  a.dot red, 20
  a.fd 100
# Looks like our turtle ran out of gas.

Tasks
