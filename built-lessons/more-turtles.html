Making More Turtles

b = new Turtle() # Make a new turtle, "b"
# Prefix commands with "b." to tell "b" to do them
b.pen red
b.fd 100
b.rt 90
b.fd 100

# Commands without "b." still affect the original turtle
pen red
bk 100
lt 90
fd 100
