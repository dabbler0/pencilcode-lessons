First Class Functions

Functions and command names can be passed around as arguments!

square = (n) ->
  for [1..4]
    fd n
    rt 90

makeThree = (f) -> # f will be a function
  f 50 # Use whatever function "f" is
  fd 100
  f 50
  fd 100
  f 50

pen red
makeThree square # Call makeThree with f = square
